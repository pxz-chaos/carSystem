# 邮箱验证码配置示例
# QQ 邮箱：
# SMTP_HOST=smtp.qq.com
# SMTP_PORT=465
# SMTP_USE_SSL=1
# SMTP_USERNAME=你的QQ邮箱@qq.com
# SMTP_PASSWORD=QQ邮箱授权码，不是QQ登录密码
# SMTP_FROM_EMAIL=你的QQ邮箱@qq.com

EMAIL_PROVIDER=smtp
SMTP_HOST=smtp.qq.com
SMTP_PORT=465
SMTP_USE_SSL=1
SMTP_USE_TLS=0
SMTP_USERNAME=你的QQ邮箱@qq.com
SMTP_PASSWORD=你的邮箱授权码
SMTP_FROM_NAME=车辆管理系统
SMTP_FROM_EMAIL=你的QQ邮箱@qq.com
EMAIL_CODE_TTL_SECONDS=300
EMAIL_SEND_COOLDOWN_SECONDS=60
EMAIL_CODE_LENGTH=6
